Binary Search
import time
start = time.time()
def binary(list1, low, high, key):
    mid = (high + low) // 2
    if high < low:
        return -1
    if list1[mid] < key:
        return binary(list1, mid + 1, high, key)
    elif list1[mid] > key:
        return binary(list1, low, mid - 1, key)
    else:
        return mid
list1 = [4, 8, 10, 15, 17, 19, 20]
high = len(list1) - 1
low = 0
print(list1)
key = int(input("Enter any number: "))
res = binary(list1, low, high, key)
if res != -1:
    print("Element is present at index", "[", res, "]")
else:
    print("Element not found")
end = time.time()
print("Time complexity", end - start)
Bubble Sort
import time
start = time.time()
def bubble(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
arr = [1, 44, 2, 8, 5, 3, 4]
bubble(arr)
print("Sorted array is:")
for i in range(len(arr)):
    print("%d" % arr[i], end=" ")
end = time.time()
print("\nThe complexity of the program is", end - start)

Merge Sort
import time
start = time.time()

def mergesort(list1):
    if len(list1) > 1:
        mid = len(list1) // 2
        left = list1[:mid]
        right = list1[mid:]
        mergesort(left)
        mergesort(right)
        i = j = k = 0
        while i < len(left) and j < len(right):
            if left[i] < right[j]:
                list1[k] = left[i]
                i = i + 1
                k = k + 1
            else:
                list1[k] = right[j]
                j = j + 1
                k = k + 1
        while i < len(left):
            list1[k] = left[i]
            i = i + 1
            k = k + 1
        while j < len(right):
            list1[k] = right[j]
            j = j + 1
            k = k + 1

Quick Sort
import time
start = time.time()

def partition(left, right, nums):
    pivot, ptr = nums[right], left
    for i in range(left, right):
        if nums[i] <= pivot:
            nums[i], nums[ptr] = nums[ptr], nums[i]
            ptr += 1
    nums[ptr], nums[right] = nums[right], nums[ptr]
    return ptr

def quicksort(left, right, nums):
    if len(nums) == 1:
        return nums
    if left < right:
        pi = partition(left, right, nums)
        quicksort(left, pi - 1, nums)
    return nums

list1 = [30, 50, 90, 10, 80, 40, 70]
print("Unsorted:", list1)
quicksort(0, len(list1) - 1, list1)
print("Sorted:", list1)

end = time.time()
print("Time complexity of the program is", end - start)

def towerofhanio(n, source, destination, auxiliary):
    if n == 1:

Bracket Balance
import time
start = time.time()
def check_balance(stack):
    count = 0
    ans = False
    for i in stack:
        if i == "(" or i == "{" or i == "[":
            count += 1
        elif i == ")" or i == "}" or i == "]":
            count -= 1
        if count < 0:
            return ans
        if count == 0:
            return not ans
    return ans
stack = input("Enter a string of brackets: ")
print("Given string is balanced:", check_balance(stack))
end = time.time()
print("The time complexity of the program is", end - start)

    '70': ['50', '90'],

def bfs(visited, graph, node):
    visited.append(node)
    queue.append(node)
    while queue:
        m = queue.pop(0)
        print(m, end=" ")
        for neighbour in graph[m]:
            if neighbour not in visited:
                visited.append(neighbour)
                queue.append(neighbour)

import time
start = time.time()
queue = []
def enqueue():
    e = input("Enter the element")
    queue.append(e)
    print(e, "is added to queue")
def dequeue():
    if not queue:
        print("queue is empty")
    else:
        e = queue.pop(0)
        print("removed element:", e)
def display():
    print(queue)
while True:
    print("selection the operation 1.add 2.remove 3.show 4.quit")
    choice = int(input())
    if choice == 1:
        enqueue()
    elif choice == 2:
        dequeue()
    elif choice == 3:
        display()
    elif choice == 4:
        break
    else:
        print("Invalid choice. Please select a valid operation.")
end = time.time()
print("the time is", end - start)
